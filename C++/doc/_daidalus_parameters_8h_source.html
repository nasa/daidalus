972 template <typename T>T parameter_data(const ParameterData& p, const std::string& key, Function<const std::string&,T> &f){
973  T default_val = f.apply(key);
974  if (p.contains(key)) {
975  return default_val;
976  }
977  DaidalusParameters::aliasestype::const_iterator find_ptr = DaidalusParameters::getAliases().find(key);
978  if (find_ptr == DaidalusParameters::getAliases().end()) {
979  return default_val;
980  }
981  std::vector<std::string>::const_iterator alias_ptr;
982  for (alias_ptr = find_ptr->second.begin(); alias_ptr != find_ptr->second.end(); ++alias_ptr) {
983  if (p.contains(*alias_ptr)) {
984  return f.apply(*alias_ptr);
985  }
986  }
987  return default_val;
988 }
